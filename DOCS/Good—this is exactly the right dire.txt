```rust
fn decode_chunk(bytes: &Bytes) -> impl Iterator<Item = f32> + '_ {
```

## ✅ Step 2.2 — Arrow builder (incremental)

```rust
use arrow_array::builder::Float32Builder;

struct ArrowAccumulator {
    builder: Float32Builder,
}

impl ArrowAccumulator {
    fn new(capacity: usize) -> Self {
        Self {
            builder: Float32Builder::with_capacity(capacity),
        }
    }

    fn push_chunk(&mut self, bytes: &Bytes) {
        for v in decode_chunk(bytes) {
            self.builder.append_value(v);
        }
    }

    fn finish(self) -> arrow_array::Float32Array {
        self.builder.finish()
    }
}
```

## ✅ Step 2.3 — Consume `ByteStream` properly

Here’s the **critical piece**:

```rust
use futures::StreamExt;
use aws_sdk_s3::primitives::ByteStream;

async fn consume_object(
    mut stream: ByteStream,
    acc: &mut ArrowAccumulator,
) {
    while let Some(chunk) = stream.next().await {
        let bytes = chunk.expect("stream error");
        acc.push_chunk(&bytes); // 🔥 no copies
    }
}
```

## ✅ Step 2.4 — Full dataset ingestion (STREAMING)

```rust
use futures::StreamExt;
use arrow_array::{ArrayRef, Float32Array};
use arrow_schema::{Schema, Field, DataType};
use arrow::record_batch::RecordBatch;
use std::sync::Arc;

pub async fn ingest_streaming<S>(
    mut objects: S,
) -> RecordBatch
where
    S: futures::Stream<Item = (String, ByteStream)> + Unpin,
{
    let mut acc = ArrowAccumulator::new(1024);

    while let Some((_path, stream)) = objects.next().await {
        consume_object(stream, &mut acc).await;
    }

    let array: ArrayRef = Arc::new(acc.finish());

    let schema = Arc::new(Schema::new(vec![
        Field::new("value", DataType::Float32, false),
    ]));

    RecordBatch::try_new(schema, vec![array]).unwrap()
}
```

# ⚡ 3. ADD CONCURRENCY (without breaking streaming)

Right now: sequential.

Upgrade:

```rust
use futures::{StreamExt, stream};

pub async fn ingest_parallel<S>(
    objects: S,
) -> RecordBatch
where
    S: futures::Stream<Item = (String, ByteStream)> + Unpin,
{
    let mut acc = ArrowAccumulator::new(1024);

    objects
        .map(|(_, stream)| async move {
            let mut local = Vec::new();

            let mut s = stream;
            while let Some(chunk) = s.next().await {
                let bytes = chunk.ok()?;
                local.extend(decode_chunk(&bytes));
            }

            Some(local)
        })
        .buffer_unordered(16)
        .filter_map(|x| async move { x })
        .for_each(|vals| {
            for v in vals {
                acc.builder.append_value(v);
            }
            futures::future::ready(())
        })
        .await;

    let array = Arc::new(acc.finish()) as ArrayRef;

    let schema = Arc::new(Schema::new(vec![
        Field::new("value", DataType::Float32, false),
    ]));

    RecordBatch::try_new(schema, vec![array]).unwrap()
}
```

### This:


👉 introduces a **small buffer per object**, but:

* keeps parallelism
* avoids locking the builder

If you want **true zero-copy + parallel**, you need:
👉 per-thread builders + merge (more complex, can do next)
