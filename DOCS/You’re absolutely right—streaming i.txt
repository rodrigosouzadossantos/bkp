## `storage/src/api.rs`

```rust
use bytes::Bytes;
use crossbeam::channel::{bounded, Receiver};
use std::thread;

use crate::providers::{get_providers, StorageProvider};

//////////////////////////////////////////////////////
// STREAMING API (HPC)
//////////////////////////////////////////////////////

/// Stream bytes from dataset (multi-provider, parallel)
pub fn stream_dataset(dataset: &str) -> Receiver<Bytes> {

  let (tx, rx) = bounded::<Bytes>(1024);

  let providers = get_providers();

  let dataset_paths = resolve_dataset_paths(dataset);

  for provider in providers {

    let tx = tx.clone();
    let paths = dataset_paths.clone();

    thread::spawn(move || {

      for path in paths {

        if matches_provider(provider.name(), &path) {

          let stream = provider.stream(&path);

          for chunk in stream {
            if tx.send(chunk).is_err() {
              return;
            }
          }

        }

      }

    });

  }

  rx
}

//////////////////////////////////////////////////////
// INTERNAL
//////////////////////////////////////////////////////

fn resolve_dataset_paths(name: &str) -> Vec<String> {

  match name {
    "images" => vec![
      "./data/images".to_string(),
      "s3://ml-bucket/images".to_string(),
    ],
    _ => panic!("dataset not found"),
  }
}

fn matches_provider(provider: &str, path: &str) -> bool {

  match provider {
    "local" => path.starts_with("./") || path.starts_with("/"),
    "s3" => path.starts_with("s3://"),
    _ => false,
  }
}
```

```rust
use bytes::Bytes;
use std::fs::File;
use std::io::{BufReader, Read};

use crate::providers::StorageProvider;

pub struct LocalProvider;

impl LocalProvider {
  pub fn new() -> Self {
    Self
  }
}

impl StorageProvider for LocalProvider {

  fn name(&self) -> &'static str {
    "local"
  }

  fn stream(&self, path: &str) -> Box<dyn Iterator<Item = Bytes> + Send> {

    let file = File::open(path).expect("open failed");
    let mut reader = BufReader::new(file);

    let iter = std::iter::from_fn(move || {
      let mut buf = vec![0u8; 64 * 1024];

      match reader.read(&mut buf) {
        Ok(0) => None,
        Ok(n) => {
          buf.truncate(n);
          Some(Bytes::from(buf))
        }
        Err(_) => None,
      }
    });

    Box::new(iter)
  }
}
```

## `dataset/src/api.rs`

```rust
use arrow_array::{Float32Array, RecordBatch};
use arrow_schema::{DataType, Field, Schema};

use std::sync::Arc;

use crossbeam::channel::Receiver;
use bytes::Bytes;

use storage::api::stream_dataset;

//////////////////////////////////////////////////////
// STREAMING INGEST → ARROW
//////////////////////////////////////////////////////

pub fn ingest_stream(dataset: &str) -> Vec<RecordBatch> {

  let rx: Receiver<Bytes> = stream_dataset(dataset);

  let mut batches = Vec::new();

  let mut builder = Vec::<f32>::with_capacity(1_000_000);

  while let Ok(chunk) = rx.recv() {

    decode_into(&chunk, &mut builder);

    if builder.len() >= 1_000_000 {

      batches.push(build_batch(&mut builder));
      builder.clear();

    }
  }

  if !builder.is_empty() {
    batches.push(build_batch(&mut builder));
  }

  batches
}

//////////////////////////////////////////////////////
// ZERO-COPY FRIENDLY BUILD
//////////////////////////////////////////////////////

fn build_batch(buffer: &mut Vec<f32>) -> RecordBatch {

  let array = Float32Array::from(buffer.clone()); // still copy (next step: builder)

  let schema = Arc::new(Schema::new(vec![
    Field::new("value", DataType::Float32, false)
  ]));

  RecordBatch::try_new(schema, vec![Arc::new(array)]).unwrap()
}

//////////////////////////////////////////////////////
// SIMD-FRIENDLY DECODE
//////////////////////////////////////////////////////

fn decode_into(bytes: &Bytes, out: &mut Vec<f32>) {

  out.extend(
    bytes.iter().map(|b| *b as f32 / 255.0)
  );
}
```

```rust
use pyo3::prelude::*;
use arrow::ffi::{FFI_ArrowArray, FFI_ArrowSchema};

use crate::api::ingest_stream;

#[pyclass]
pub struct Dataset;

#[pymethods]
impl Dataset {

  #[staticmethod]
  pub fn ingest_arrow(py: Python<'_>, name: &str) -> PyObject {

    let batches = ingest_stream(name);

    let batch = batches.into_iter().next().unwrap();
    let array = batch.column(0).clone();

    let (ffi_array, ffi_schema) = arrow::ffi::export_array_to_c(array);

    let capsule_array = PyCapsule::new(py, ffi_array, None).unwrap();
    let capsule_schema = PyCapsule::new(py, ffi_schema, None).unwrap();

    let pyarrow = py.import("pyarrow").unwrap();

    let array = pyarrow
      .call_method1("_import_from_c", (capsule_array, capsule_schema))
      .unwrap();

    array.into()
  }
}
```
