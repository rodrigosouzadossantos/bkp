# ☁️ 2. `storage/src/providers/s3.rs`

```rust
use aws_sdk_s3::{Client, primitives::ByteStream};
use futures::{stream, StreamExt};
use futures::stream::BoxStream;

use crate::providers::StorageProvider;

pub struct S3Provider {
    pub client: Client,
}

impl StorageProvider for S3Provider {
    fn name(&self) -> &str {
        "s3"
    }

    fn roots(&self) -> Vec<String> {
        vec!["s3://".to_string()]
    }

    fn list(&self, _path: &str) -> Vec<String> {
        vec![] // keep your existing impl if needed
    }

    fn read_dir_stream(
        &self,
        path: &str,
    ) -> BoxStream<'static, (String, ByteStream)> {

        let (bucket, prefix) = parse_s3_path(path);
        let client = self.client.clone();

        let key_stream = stream::unfold(
            (client.clone(), bucket.clone(), prefix.unwrap_or_default(), None),
            |(client, bucket, prefix, token)| async move {
                let resp = client
                    .list_objects_v2()
                    .bucket(&bucket)
                    .prefix(&prefix)
                    .set_continuation_token(token)
                    .send()
                    .await
                    .ok()?;

                let next = resp.next_continuation_token.clone();

                let keys: Vec<String> = resp
                    .contents
                    .unwrap_or_default()
                    .into_iter()
                    .filter_map(|o| o.key)
                    .collect();

                Some((stream::iter(keys), (client, bucket, prefix, next)))
            },
        )
        .flatten();

        let object_stream = key_stream
            .map(move |key| {
                let client = client.clone();
                let bucket = bucket.clone();

                async move {
                    let resp = client
                        .get_object()
                        .bucket(&bucket)
                        .key(&key)
                        .send()
                        .await
                        .ok()?;

                    Some((key, resp.body))
                }
            })
            .buffer_unordered(32)
            .filter_map(|x| async move { x });

        Box::pin(object_stream)
    }
}
```

# 🧠 4. `dataset/src/api.rs` (FULL REWRITE)

```rust
use std::sync::Arc;

use arrow_array::{ArrayRef, Float32Array, builder::Float32Builder};
use arrow_schema::{DataType, Field, Schema};
use arrow::record_batch::RecordBatch;

use bytes::Bytes;
use futures::StreamExt;
use aws_sdk_s3::primitives::ByteStream;

use storage::api as storage_api;

//////////////////////////////////////////////////////
// DECODER
//////////////////////////////////////////////////////

fn decode_chunk(bytes: &Bytes) -> impl Iterator<Item = f32> + '_ {
    bytes.iter().map(|b| *b as f32 / 255.0)
}

//////////////////////////////////////////////////////
// ARROW ACCUMULATOR
//////////////////////////////////////////////////////

struct ArrowAccumulator {
    builder: Float32Builder,
}

impl ArrowAccumulator {
    fn new() -> Self {
        Self {
            builder: Float32Builder::new(),
        }
    }

    fn push(&mut self, bytes: &Bytes) {
        for v in decode_chunk(bytes) {
            self.builder.append_value(v);
        }
    }

    fn finish(self) -> Float32Array {
        self.builder.finish()
    }
}

//////////////////////////////////////////////////////
// STREAMING INGEST
//////////////////////////////////////////////////////

pub async fn ingest_streaming(path: &str) -> RecordBatch {
    let mut stream = storage_api::read_dir_stream(path);

    let mut acc = ArrowAccumulator::new();

    while let Some((_path, mut obj_stream)) = stream.next().await {
        while let Some(chunk) = obj_stream.next().await {
            let bytes = chunk.expect("stream error");
            acc.push(&bytes);
        }
    }

    let array: ArrayRef = Arc::new(acc.finish());

    let schema = Arc::new(Schema::new(vec![
        Field::new("value", DataType::Float32, false),
    ]));

    RecordBatch::try_new(schema, vec![array]).unwrap()
}
```
